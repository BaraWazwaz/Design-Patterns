20/// @brief data that are serializable using an iostream
21template <typename Type>
22concept Streamable = requires(std::ostream& os,
23 std::istream& is,
24 Type vm,
25 Type const& vc)
26{
27 { os << vc };
28 { is >> vm };
29};
