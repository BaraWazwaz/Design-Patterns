14/// @brief a proper singleton should not be publically constructable or moved
15template <typename T>
16concept ProperSingleton = !std::is_default_constructible_v<T> &&
17 !std::is_copy_constructible_v<T> &&
18 !std::is_move_constructible_v<T> &&
19 !std::is_copy_assignable_v<T> &&
20 !std::is_move_assignable_v<T>;
21
22/// @brief Singleton wrapper for safe access of single instance classes
23/// @tparam ValueType needs to have a private or protected default constructor,
24/// needs to have nitron::Singleton<T> as a friend
25template <typename ValueType>
26class Singleton
27{
28public:
29 /// - `first` is the instance reference
30 /// - `second` is mutex lock guard object for thread safey
31 using AccessToken = std::pair<ValueType&, std::lock_guard<std::mutex>>;
32
33 static_assert(ProperSingleton<ValueType>,
34 "Wrapped type is NOT suitable to be a Singleton.");
35
36 Singleton() = delete;
37 Singleton(Singleton const&) = delete;
38 Singleton(Singleton&&) = delete;
39 Singleton& operator=(Singleton const&) = delete;
40 Singleton& operator=(Singleton&&) = delete;
41
42 /// @brief acquiring @ref AccessToken for instance
43 static AccessToken get();
44};
